# List集合去重


``` java
   @Override
   	public List<ThirdpartyPort> findThirdpartyPorts(ThirdpartyPortFilterRequest filter) {
   		HashSet<String> setTempList = new HashSet<String>();
   		List<ThirdpartyPort> thirdPartPortTypeList = new ArrayList<>();
   		List<ThirdpartyPort> thirdPartPortList = new Executor(thirdpartyportRepos).findAll(filter);
   		if (CollectionUtils.isEmpty(thirdPartPortList)) {
   			throw new ParameterException();
   		}
   		for (ThirdpartyPort thirdPartyPort : thirdPartPortList) {
   			if (setTempList.contains(thirdPartyPort.getPortType())) {
   				continue;
   			}
   			setTempList.add(thirdPartyPort.getPortType());
   			thirdPartPortTypeList.add(thirdPartyPort);
   		}
   		return thirdPartPortTypeList;
   	}
```
